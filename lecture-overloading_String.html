========복습========

생성자

인스턴스를 만들 때 마지막 단계

클래스에 정의한 변수 생성되고 그 다음에 생성자 메소드 호출

호출한 메소드 내의 내용을 하나씩 실행하면서 생성자 메소드를 마무리

다른 메소드들이랑 다르게 void(반환자료형이 없고, 리턴도 없음)

생성자가 하나도 없다면 기본 생성자를 시스템이서 만들어주고 생성자가 있다면 기본 생성자를 만들어 주지 않는다

우리가 정의한 생성자의 모양대로만 new 인스턴스를 할 수 있다

  • 만약에 생성자에 파라미터가 2개 있는 생성자를 만들었다

    • 이 다음 부터는 클래스를 인스턴스화 할 때 무조건 인자를 2개 넣어야한다!

    • 인자를 넣지 않는 인스턴스는 더 이상 만들지 못함

과정

  • 인스턴스 만들기(new를 한다) » 생성자 메소드가 1회 호출된다

overloading

메소드를 중복해서 정의하는 것

메소드 시그니처

  • 메소드를 정의할 때 쓰는 파라미터 요소(리스트)

메소드 시그니처가 다르면 같은 이름을 가질 수 있음, 반환타입과는 무관

timer 깔끔하게 구현해보기

class Timer22 {

	String clock(int a) {
		if (a >= 0)
			return "째깍째깍 현재시간은 " + (a / 3600) + "시 " + a % 3600 / 60 + "분" + a % 60 + "초 입니다!";
		return "Invalid Value";
	}

	String clock(int a, int b) {
		if (a >= 0 && b >= 0)
			return clock(a * 60 + b);
		return "Invalid Value";
	}

	String clock(String a, int b) {
		if (!a.equals("윤년") || b < 0)
			return "Invalid Value";
		if ((b % 4 == 0 && b % 100 != 0) || b % 400 == 0)
			return b + "년은 윤년입니다";
		return b + "년은 윤년이 아닙니다";
	}
}

String

문자열을 담는 자료형

package java.lang 에 있는 클래스

String str1 = new String(); //파라미터의 정의가 없는 생성자

String str2 = new String("hello"); //파라미터의 정의가 있는 생성자

String str3 = “안녕”; //자바에서 보편적으로 사용하는 코드이기 때문에 예외적으로 문법을 무시한 케이스

str.length(); //배열이라 길이를 구할 수 있음

str.charAt(int); //특정 인덱스 번호에 있는 문자를 반환

str1.compareTo(str2); //특정 문자열 비교하는 문장 차이나는 개수만큼 출력, 대소문자도 구분

str1.concat(str2); //문자열을 더하는 메소드, 원본 데이터에 영향 없음

indexOf(str); //문자열에서 문자 검색, 없을 경우 -1

toLowerCase(str); // 소문자 변환
toUpperCase(str); // 대문자 변환

substring(str); //문자열 자르기 (인자 하나 begin 인자 둘 begin end)

split(str); //문자열 분리하기, 인자(분리하는 기준)

replace(str1,str2); //문자열 중에 문자를 str1 => str2 바꿔버리는 것

str.trim(); //앞 뒤로 공백을 제거

String.valueOf(str); //모든 자료형을 문자열로 바꿔줌
